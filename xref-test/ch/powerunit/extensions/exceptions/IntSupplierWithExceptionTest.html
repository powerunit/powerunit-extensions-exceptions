20  package ch.powerunit.extensions.exceptions;
21  
22  import ch.powerunit.Test;
23  import ch.powerunit.TestSuite;
24  
25  @SuppressWarnings("squid:S2187") // Sonar doesn't under that it is really a test
26  public class IntSupplierWithExceptionTest implements TestSuite {
27  
28  	@Test
29  	public void testFailing() {
30  		assertWhen((x) -> IntSupplierWithException.failing(Exception::new).getAsInt())
31  				.throwException(instanceOf(Exception.class));
32  	}
33  
34  	@Test
35  	public void testCheckedNoException() {
36  		assertThat(IntSupplierWithException.unchecked(() -> 1).getAsInt()).is(1);
37  	}
38  
39  	@Test
40  	public void testCheckedException() {
41  		assertWhen((x) -> IntSupplierWithException.unchecked(() -> {
42  			throw new Exception();
43  		}).getAsInt()).throwException(instanceOf(WrappedException.class));
44  	}
45  
46  	@Test
47  	public void testCheckedExceptionHandler() {
48  		assertWhen((x) -> IntSupplierWithException.unchecked(() -> {
49  			throw new Exception();
50  		}, RuntimeException::new).getAsInt()).throwException(instanceOf(RuntimeException.class));
51  	}
52  
53  	@Test
54  	public void testLiftedNoException() {
55  		assertThat(IntSupplierWithException.lifted(() -> 1).getAsInt()).is(1);
56  	}
57  
58  	@Test
59  	public void testLiftedException() {
60  		assertThat(IntSupplierWithException.lifted(() -> {
61  			throw new Exception();
62  		}).getAsInt()).is(0);
63  	}
64  
65  	@Test
66  	public void testIgnoredNoException() {
67  		assertThat(IntSupplierWithException.ignored(() -> 2).getAsInt()).is(2);
68  	}
69  
70  	@Test
71  	public void testIgnoredException() {
72  		assertThat(IntSupplierWithException.ignored(() -> {
73  			throw new Exception();
74  		}).getAsInt()).is(0);
75  	}
76  
77  	@Test
78  	public void testIgnoredDefaultNoException() {
79  		assertThat(IntSupplierWithException.ignored(() -> 2, 1).getAsInt()).is(2);
80  	}
81  
82  	@Test
83  	public void testIgnoredDefaultException() {
84  		assertThat(IntSupplierWithException.ignored(() -> {
85  			throw new Exception();
86  		}, 1).getAsInt()).is(1);
87  	}
88  
89  }
